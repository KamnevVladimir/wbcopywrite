# ==============================================

# Application

LOG_LEVEL=debug

# Server
SERVER_HOST=0.0.0.0
SERVER_PORT=8080

# Database (Local PostgreSQL via Docker)
DATABASE_HOST=localhost
DATABASE_PORT=5432

# Telegram Bot
# Получить: https://t.me/botfather -> /newbot
TELEGRAM_BOT_TOKEN=your_telegram_bot_token_here

# Claude API
# Получить: https://console.anthropic.com/ -> API Keys

TRIBUTE_API_KEY=your_tribute_api_key
TRIBUTE_SECRET=your_tribute_secret_key
TRIBUTE_API_URL=https://api.tribute.to/v1

# GlitchTip / Sentry (Error Tracking & Monitoring)
# Получить: https://glitchtip.com/ -> New Project -> DSN
# Формат: https://[key]@glitchtip.com/[project]
GLITCHTIP_DSN=https://your_key@glitchtip.com/your_project
# Или для Sentry:
# SENTRY_DSN=https://your_key@sentry.io/your_project

# Rate Limiting (лимиты генераций)
RATE_LIMIT_FREE=3
RATE_LIMIT_STARTER=30
RATE_LIMIT_BUSINESS=150
RATE_LIMIT_PRO=500
